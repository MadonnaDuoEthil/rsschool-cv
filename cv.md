```
function bouncer(arr) {
  let copyArr = arr.slice();
  let resultArr = [];
  for (let i = 0; i < copyArr.length; i++) {
    if (Boolean(copyArr[i]) === true) {
      resultArr.push(copyArr[i]);
    }
  }
  return resultArr;
}
```
